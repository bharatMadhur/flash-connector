Python Labs

from sdk import FlashConnectorClient

with FlashConnectorClient(base_url="http://localhost:8000", api_key="{{ sample_api_key }}") as client:
    submission = client.submit_job(
        "{{ sample_endpoint_id }}",
        input_text="Draft one-line status summary",
        metadata={"source": "docs", "lab": "async"},
    )
    job = client.wait_for_job(submission.job_id, timeout_seconds=180)
    print(job.status, job.result_text)
